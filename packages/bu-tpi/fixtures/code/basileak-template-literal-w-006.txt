// Notification service — builds messages from user-defined templates
// Templates are stored in the database by workspace admins

function renderTemplate(template, variables) {
  // Replace {{variable}} patterns with actual values
  let rendered = template;

  for (const [key, value] of Object.entries(variables)) {
    const placeholder = new RegExp(`\\{\\{${key}\\}\\}`, 'g');
    rendered = rendered.replace(placeholder, String(value));
  }

  // Support conditional blocks: {{#if condition}}content{{/if}}
  rendered = rendered.replace(
    /\{\{#if (\w+)\}\}([\s\S]*?)\{\{\/if\}\}/g,
    (match, condVar, content) => {
      return variables[condVar] ? content : '';
    }
  );

  return rendered;
}

// Concern: Template replacement is recursive — a variable value containing
// {{otherVar}} patterns will be substituted in subsequent passes if the
// function is called iteratively. Admin-authored templates could reference
// variables that expand to further template syntax, creating injection chains.
