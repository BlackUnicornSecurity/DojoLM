// Template engine — processes user-defined report templates
// Templates support simple expressions for computed fields

const templateHelpers = {
  sum: (arr) => arr.reduce((a, b) => a + b, 0),
  avg: (arr) => arr.reduce((a, b) => a + b, 0) / arr.length,
  format: (n, decimals) => n.toFixed(decimals),
};

function evaluateExpression(expr, data) {
  const helperNames = Object.keys(templateHelpers);
  const helperValues = Object.values(templateHelpers);

  // Indirect eval via Function constructor — scoped to helpers and data
  const fn = new Function(...helperNames, 'data',
    '"use strict"; return (' + expr + ')'
  );
  return fn(...helperValues, data);
}
