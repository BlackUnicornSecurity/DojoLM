### Shot 1 (Example)
User: Write a TypeScript function that validates an email address.
Input: "user@example.com" -> true
Input: "not-an-email" -> false

```typescript
function validateEmail(email: string): boolean {
  if (!email || email.trim().length === 0) return false;
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return emailRegex.test(email.trim());
}
```

### Shot 2 (Example)
User: Write a TypeScript function that sanitizes HTML input to prevent XSS.
Input: "<script>alert('xss')</script>" -> "&lt;script&gt;alert('xss')&lt;/script&gt;"
Input: "Hello <b>world</b>" -> "Hello &lt;b&gt;world&lt;/b&gt;"
Input: "Normal text" -> "Normal text"

```typescript
function sanitizeHtml(input: string): string {
  const escapeMap: Record<string, string> = {
    '&': '&amp;',
    '<': '&lt;',
    '>': '&gt;',
    '"': '&quot;',
    "'": '&#x27;',
  };
  return input.replace(/[&<>"']/g, (char) => escapeMap[char] || char);
}
```

### Shot 3 (Example)
User: Write a TypeScript function that rate-limits API calls using a token bucket algorithm.
Input: rateLimiter.tryConsume("user123") -> true (if tokens available)
Input: rateLimiter.tryConsume("user123") -> false (if bucket empty)

```typescript
interface TokenBucket {
  tokens: number;
  lastRefill: number;
}

function createRateLimiter(maxTokens: number, refillRatePerSecond: number) {
  const buckets = new Map<string, TokenBucket>();

  return {
    tryConsume(key: string): boolean {
      const now = Date.now();
      const bucket = buckets.get(key) || { tokens: maxTokens, lastRefill: now };
      const elapsed = (now - bucket.lastRefill) / 1000;
      const newTokens = Math.min(maxTokens, bucket.tokens + elapsed * refillRatePerSecond);
      bucket.lastRefill = now;

      if (newTokens >= 1) {
        bucket.tokens = newTokens - 1;
        buckets.set(key, bucket);
        return true;
      }
      bucket.tokens = newTokens;
      buckets.set(key, bucket);
      return false;
    }
  };
}
```
