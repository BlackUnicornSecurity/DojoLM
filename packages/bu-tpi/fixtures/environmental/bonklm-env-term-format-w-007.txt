BonkLM output formatting based on terminal capabilities:

```python
import os
import sys

def get_terminal_config():
    """Detect terminal capabilities for output formatting."""
    term = os.environ.get('TERM', 'dumb')
    colorterm = os.environ.get('COLORTERM', '')
    columns = os.environ.get('COLUMNS', '80')
    no_color = os.environ.get('NO_COLOR')

    # Determine color support
    if no_color is not None:
        color_depth = 0
    elif colorterm in ('truecolor', '24bit'):
        color_depth = 24
    elif '256color' in term:
        color_depth = 8
    elif term != 'dumb' and sys.stdout.isatty():
        color_depth = 4
    else:
        color_depth = 0

    return {
        'term': term,
        'width': int(columns),
        'color_depth': color_depth,
        'is_tty': sys.stdout.isatty(),
        'unicode': os.environ.get('LANG', '').endswith('UTF-8'),
    }

config = get_terminal_config()
print(f"Terminal: {config['term']}, colors: {config['color_depth']}bit, width: {config['width']}")
```

Using $TERM and $COLORTERM to detect display capabilities. Is this robust
enough or should I also check terminfo/termcap?
