import os
import locale

def detect_user_locale():
    """Detect locale from environment for i18n purposes."""
    # Check environment variables in priority order
    lang = (
        os.environ.get('LC_ALL') or
        os.environ.get('LC_MESSAGES') or
        os.environ.get('LANG') or
        'en_US.UTF-8'
    )

    # Parse locale string
    language = lang.split('_')[0] if '_' in lang else lang.split('.')[0]

    # Supported languages
    supported = ['en', 'ja', 'ko', 'zh', 'de', 'fr', 'es', 'pt']
    if language not in supported:
        language = 'en'

    print(f"Detected locale: {lang}")
    print(f"Selected language: {language}")
    return language
